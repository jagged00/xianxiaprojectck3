# Cultivation Character Initialization Events
# Path: events/cultivation_character_initialization_events.txt
# Based on ACTUAL Carnalitas implementation

namespace = cultivation_character_initialization

# Initialize all living characters (game start)
cultivation_character_initialization.0001 = {
	type = character_event
	scope = none
	hidden = yes

	immediate = {
		every_living_character = {
			trigger_event = cultivation_character_initialization.0002
			trigger_event = cultivation_character_initialization.0003
			trigger_event = cultivation_character_initialization.0004
		}
	}
}

# Ensure character has one active meridian trait
cultivation_character_initialization.0002 = {
	type = character_event
	hidden = yes

	trigger = {
		is_alive = yes
		NOT = { has_active_meridians = yes }
	}

	immediate = {
		random_list = {
			# Novel-style rarity curve:
			# poor ~85%, normal ~12%, good ~2.8%, excellent ~0.19%, heavenly ~0.0001%
			850000 = { add_trait = poor_meridians }
			120000 = { add_trait = normal_meridians }
			28000 = { add_trait = good_meridians }
			1900 = { add_trait = excellent_meridians }
			1 = { add_trait = heavenly_meridians }
		}
	}
}

# Ensure every adult has a cultivation realm (qi gathering+) like education traits.
cultivation_character_initialization.0003 = {
	type = character_event
	hidden = yes

	trigger = {
		is_alive = yes
		is_adult = yes
		has_active_meridians = yes
		NOT = { is_cultivator = yes }
	}

	immediate = {
		add_trait = cultivation_qi_gathering
	}
}

# Ensure AI adults with realms join a cultivation sect automatically.
cultivation_character_initialization.0004 = {
	type = character_event
	hidden = yes

	trigger = {
		is_alive = yes
		is_ai = yes
		is_adult = yes
		has_active_meridians = yes
		is_cultivator = yes
		NOT = {
			faith = {
				OR = {
					has_doctrine = tenet_orthodox_cultivation
					has_doctrine = tenet_unorthodox_cultivation
					has_doctrine = tenet_demonic_cultivation
					has_doctrine = tenet_vagrant_cultivation
				}
			}
		}
	}

	immediate = {
		if = {
			limit = { has_trait = poor_meridians }
			random_list = {
				55 = { set_character_faith = faith:lone_wolf_wanderers }
				25 = { set_character_faith = faith:vagrant_blade_brotherhood }
				20 = { set_character_faith = faith:thousand_treasures_pavilion }
			}
		}
		else_if = {
			limit = { has_trait = normal_meridians }
			random_list = {
				35 = { set_character_faith = faith:mount_hua_sect }
				20 = { set_character_faith = faith:wudang_sect }
				20 = { set_character_faith = faith:jade_chamber_sect }
				15 = { set_character_faith = faith:lone_wolf_wanderers }
				10 = { set_character_faith = faith:blood_moon_demonic_sect }
			}
		}
		else_if = {
			limit = { has_trait = good_meridians }
			random_list = {
				28 = { set_character_faith = faith:mount_hua_sect }
				20 = { set_character_faith = faith:wudang_sect }
				20 = { set_character_faith = faith:heavenly_sword_pavilion }
				12 = { set_character_faith = faith:jade_chamber_sect }
				10 = { set_character_faith = faith:medicine_king_valley }
				10 = { set_character_faith = faith:crimson_moon_blade_hall }
			}
		}
		else_if = {
			limit = { has_trait = excellent_meridians }
			random_list = {
				30 = { set_character_faith = faith:mount_hua_sect }
				25 = { set_character_faith = faith:heavenly_sword_pavilion }
				20 = { set_character_faith = faith:wudang_sect }
				15 = { set_character_faith = faith:ten_thousand_swords_valley }
				10 = { set_character_faith = faith:divine_beast_mountain }
			}
		}
		else_if = {
			limit = { has_trait = heavenly_meridians }
			random_list = {
				35 = { set_character_faith = faith:heavenly_sword_pavilion }
				30 = { set_character_faith = faith:ten_thousand_swords_valley }
				20 = { set_character_faith = faith:mount_hua_sect }
				15 = { set_character_faith = faith:wudang_sect }
			}
		}
	}
}
