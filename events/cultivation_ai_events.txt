namespace = cultivation_ai

# Startup bootstrap so AI cultivation is never player-gated.
cultivation_ai.0000 = {
    type = character_event
    hidden = yes

    immediate = {
        every_living_character = {
            limit = {
                is_ai = yes
                is_adult = yes
                NOT = { has_trait = incapable }
            }
            trigger_event = cultivation_ai.0001
            trigger_event = cultivation_ai.0002
        }
    }

    option = { name = OK }
}

# AI auto-cultivation progression pulse (meridian-driven).
cultivation_ai.0001 = {
    type = character_event
    hidden = yes

    trigger = {
        is_ai = yes
        is_alive = yes
        is_adult = yes
        NOT = { has_trait = incapable }
    }

    immediate = {
        # Ensure every AI gets exactly one meridian tier.
        if = {
            limit = { NOT = { has_active_meridians = yes } }
            random_list = {
                8500 = { add_trait = poor_meridians }
                1200 = { add_trait = normal_meridians }
                280 = { add_trait = good_meridians }
                19 = { add_trait = excellent_meridians }
                1 = { add_trait = heavenly_meridians }
            }
        }

        # Awakening chance: intentionally high so the world fills with cultivators quickly.
        if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = poor_meridians
            }
            random_list = {
                35 = { add_trait = cultivation_qi_gathering }
                65 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = normal_meridians
            }
            random_list = {
                60 = { add_trait = cultivation_qi_gathering }
                40 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = good_meridians
            }
            random_list = {
                85 = { add_trait = cultivation_qi_gathering }
                15 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = excellent_meridians
            }
            random_list = {
                97 = { add_trait = cultivation_qi_gathering }
                3 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = heavenly_meridians
            }
            add_trait = cultivation_qi_gathering
        }

        # Progression chance scaled by meridian quality (checked every monthly pulse).
        if = {
            limit = { has_trait = cultivation_qi_gathering }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                3 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                8 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                16 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                28 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                944 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_qi_refining }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                2 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                6 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                12 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                20 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                959 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_qi_building }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                2 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }

                9 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                15 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                968 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_core_formation }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_core_formation add_trait = cultivation_nascent_soul }
                1 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_core_formation add_trait = cultivation_nascent_soul }

                990 = { }

        else_if = {

# AI sect-joining pulse.
cultivation_ai.0002 = {
    type = character_event
    hidden = yes

    trigger = {
        is_ai = yes
        is_alive = yes
        is_adult = yes
        has_active_meridians = yes
        NOT = { has_character_flag = ai_recently_joined_sect }
    }

    immediate = {
        if = {
            limit = {
                NOT = {
                    faith = {
                        OR = {
                            has_doctrine = tenet_orthodox_cultivation
                            has_doctrine = tenet_unorthodox_cultivation
                            has_doctrine = tenet_demonic_cultivation
                            has_doctrine = tenet_vagrant_cultivation
                        }
                    }
                }
            }

            random_list = {
                1 = {
                    trigger = { has_trait = poor_meridians }
                    random_list = {
                        55 = { set_character_faith = faith:lone_wolf_wanderers }
                        25 = { set_character_faith = faith:vagrant_blade_brotherhood }
                        20 = { set_character_faith = faith:thousand_treasures_pavilion }
                    }
                }
                6 = {
                    trigger = { has_trait = normal_meridians }
                    random_list = {
                        35 = { set_character_faith = faith:mount_hua_sect }
                        20 = { set_character_faith = faith:wudang_sect }
                        20 = { set_character_faith = faith:jade_chamber_sect }
                        15 = { set_character_faith = faith:lone_wolf_wanderers }
                        10 = { set_character_faith = faith:blood_moon_demonic_sect }
                    }
                }
                16 = {
                    trigger = { has_trait = good_meridians }
                    random_list = {
                        28 = { set_character_faith = faith:mount_hua_sect }
                        20 = { set_character_faith = faith:wudang_sect }
                        20 = { set_character_faith = faith:heavenly_sword_pavilion }
                        12 = { set_character_faith = faith:jade_chamber_sect }
                        10 = { set_character_faith = faith:medicine_king_valley }
                        10 = { set_character_faith = faith:crimson_moon_blade_hall }
                    }
                }
                40 = {
                    trigger = { has_trait = excellent_meridians }
                    random_list = {
                        30 = { set_character_faith = faith:mount_hua_sect }
                        25 = { set_character_faith = faith:heavenly_sword_pavilion }
                        20 = { set_character_faith = faith:wudang_sect }
                        15 = { set_character_faith = faith:ten_thousand_swords_valley }
                        10 = { set_character_faith = faith:divine_beast_mountain }
                    }
                }
                70 = {
                    trigger = { has_trait = heavenly_meridians }
                    random_list = {
                        35 = { set_character_faith = faith:heavenly_sword_pavilion }
                        30 = { set_character_faith = faith:ten_thousand_swords_valley }
                        20 = { set_character_faith = faith:mount_hua_sect }
                        15 = { set_character_faith = faith:wudang_sect }
                    }
                }
                867 = { }
            }

            if = {
                limit = {
                    faith = {
                        OR = {
                            has_doctrine = tenet_orthodox_cultivation
                            has_doctrine = tenet_unorthodox_cultivation
                            has_doctrine = tenet_demonic_cultivation
                            has_doctrine = tenet_vagrant_cultivation
                        }
                    }
                }
                add_character_flag = { flag = ai_recently_joined_sect days = 1825 }
            }
        }
    }

    option = { name = OK }
}
