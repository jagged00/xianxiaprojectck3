namespace = cultivation_ai

# Startup bootstrap so AI cultivation is never player-gated.
cultivation_ai.1000 = {
    type = character_event
    hidden = yes

    immediate = {
        every_living_character = {
            limit = {
                is_ai = yes
                is_adult = yes
                NOT = { has_trait = incapable }
            }
            trigger_event = cultivation_ai.0001
            trigger_event = cultivation_ai.0002
        }
    }

    option = { name = OK }
}

# AI auto-cultivation progression pulse (meridian-driven).
cultivation_ai.0001 = {
    type = character_event
    hidden = yes

    trigger = {
        is_ai = yes
        is_alive = yes
        is_adult = yes
        NOT = { has_trait = incapable }
    }

    immediate = {
        # Ensure every AI gets exactly one meridian tier.
        if = {
            limit = { NOT = { has_active_meridians = yes } }
            random_list = {
                8500 = { add_trait = poor_meridians }
                1200 = { add_trait = normal_meridians }
                280 = { add_trait = good_meridians }
                19 = { add_trait = excellent_meridians }
                1 = { add_trait = heavenly_meridians }
            }
        }

        # Awakening chance: intentionally high so the world fills with cultivators quickly.
        if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = poor_meridians
            }
            random_list = {
                35 = { add_trait = cultivation_qi_gathering }
                65 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = normal_meridians
            }
            random_list = {
                60 = { add_trait = cultivation_qi_gathering }
                40 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = good_meridians
            }
            random_list = {
                85 = { add_trait = cultivation_qi_gathering }
                15 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = excellent_meridians
            }
            random_list = {
                97 = { add_trait = cultivation_qi_gathering }
                3 = { }
            }
        }
        else_if = {
            limit = {
                NOT = { is_cultivator = yes }
                has_trait = heavenly_meridians
            }
            add_trait = cultivation_qi_gathering
        }

        # Progression chance scaled by meridian quality (checked every monthly pulse).
        if = {
            limit = { has_trait = cultivation_qi_gathering }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                3 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                10 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                24 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                40 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_qi_gathering add_trait = cultivation_qi_refining }
                922 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_qi_refining }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                2 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                6 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                18 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                30 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_qi_refining add_trait = cultivation_qi_building }
                943 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_qi_building }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                2 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                5 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                14 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                22 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_qi_building add_trait = cultivation_core_formation }
                956 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_core_formation }
            random_list = {
                1 = { trigger = { has_trait = poor_meridians } remove_trait = cultivation_core_formation add_trait = cultivation_nascent_soul }
                1 = { trigger = { has_trait = normal_meridians } remove_trait = cultivation_core_formation add_trait = cultivation_nascent_soul }
                3 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_core_formation add_trait = cultivation_nascent_soul }
                12 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_core_formation add_trait = cultivation_nascent_soul }
                20 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_core_formation add_trait = cultivation_nascent_soul }
                963 = { }
            }
        }
        else_if = {

        else_if = {

            random_list = {
                1 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_heavenly_being add_trait = cultivation_four_axis }
                5 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_heavenly_being add_trait = cultivation_four_axis }
                10 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_heavenly_being add_trait = cultivation_four_axis }
                984 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_four_axis }
            random_list = {
                1 = { trigger = { has_trait = good_meridians } remove_trait = cultivation_four_axis add_trait = cultivation_integration }

                8 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_four_axis add_trait = cultivation_integration }
                987 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_integration }
            random_list = {
                2 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_integration add_trait = cultivation_star_shattering }
                6 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_integration add_trait = cultivation_star_shattering }

        }
        else_if = {
            limit = { has_trait = cultivation_star_shattering }
            random_list = {
                2 = { trigger = { has_trait = excellent_meridians } remove_trait = cultivation_star_shattering add_trait = cultivation_sacred_vessel }
                5 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_star_shattering add_trait = cultivation_sacred_vessel }

                998 = { }
            }
        }
        else_if = {
            limit = { has_trait = cultivation_entering_nirvana }
            random_list = {
                2 = { trigger = { has_trait = heavenly_meridians } remove_trait = cultivation_entering_nirvana add_trait = cultivation_true_immortal }
                998 = { }
            }
        }

        cultivation_ai_autogrant_realm_perks_effect = yes
    }

    option = { name = OK }
}

# AI sect-joining pulse.
cultivation_ai.0002 = {
    type = character_event
    hidden = yes

    trigger = {
        is_ai = yes
        is_alive = yes
        is_adult = yes
        has_active_meridians = yes
        is_cultivator = yes
        NOT = { has_character_flag = ai_recently_joined_sect }
    }

    immediate = {
        if = {
            limit = {
                NOT = {
                    faith = {
                        OR = {
                            has_doctrine = tenet_orthodox_cultivation
                            has_doctrine = tenet_unorthodox_cultivation
                            has_doctrine = tenet_demonic_cultivation
                            has_doctrine = tenet_vagrant_cultivation
                        }
                    }
                }
            }

            cultivation_ai_join_any_sect_by_meridian_effect = yes

            add_character_flag = { flag = ai_recently_joined_sect days = 1825 }
        }
    }

    option = { name = OK }
}
