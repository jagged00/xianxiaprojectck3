namespace = cultivation_ai

# Monthly orchestrator: iterate all living AI cultivators so progression does not
# depend on implicit on_action character scope behavior.
cultivation_ai.2000 = {
    type = character_event
    scope = none
    hidden = yes

    immediate = {
        every_living_character = {
            limit = {
                is_ai = yes
                is_alive = yes
                age >= 14
                NOT = { has_trait = incapable }
                has_active_meridians = yes
            }
            trigger_event = cultivation_ai.0001
            trigger_event = cultivation_ai.0002
            trigger_event = cultivation_ai.0003
            trigger_event = cultivation_ai.0004
            trigger_event = cultivation_ai.0005
        }
    }

    option = { name = OK }
}

# AI sect-joining pulse.
cultivation_ai.0002 = {
    type = character_event
    hidden = yes

    trigger = {
        is_ai = yes
        is_alive = yes
        is_adult = yes
        has_active_meridians = yes
        is_cultivator = yes
        NOT = { has_character_flag = ai_recently_joined_sect }
    }

    immediate = {
        if = {
            limit = {
                NOT = {
                    faith = {
                        OR = {
                            has_doctrine = tenet_orthodox_cultivation
                            has_doctrine = tenet_unorthodox_cultivation
                            has_doctrine = tenet_demonic_cultivation
                            has_doctrine = tenet_vagrant_cultivation
                        }
                    }
                }
            }

            cultivation_ai_join_any_sect_by_meridian_effect = yes

            add_character_flag = { flag = ai_recently_joined_sect days = 1825 }
        }
    }

    option = { name = OK }
}

# Realm hierarchy pressure: higher-realm cultivators strongly reject weaker lieges.
cultivation_ai.0003 = {
    type = character_event
    hidden = yes

    trigger = {
        is_alive = yes
        age >= 14
        is_cultivator = yes
        exists = liege
        has_higher_cultivation_realm_than_liege = yes
        NOT = {
            OR = {
                liege = { is_parent_of = scope:root }
                liege = {
                    dynasty = scope:root.dynasty
                    is_close_or_extended_family_of = scope:root
                }
            }
        }
    }

    immediate = {
        add_opinion = {
            target = liege
            modifier = despises_weak_liege_cultivation_opinion
            years = 2
        }

        if = {
            limit = {
                is_ai = yes
                NOT = { has_character_flag = cultivation_realm_rebellion_cooldown }
            }
            add_character_flag = { flag = cultivation_realm_rebellion_cooldown days = 1825 }

            random_list = {
                40 = {
                    add_stress = -40
                }
                30 = { add_dread = 8 }
                30 = { add_prestige = 75 }
            }
        }
    }

    option = { name = OK }
}

# Personality-driven treatment of talented juniors:
# ruthless elders suppress future threats, benevolent elders invest in them.
cultivation_ai.0004 = {
    type = character_event
    hidden = yes

    trigger = {
        is_ai = yes
        is_alive = yes
        is_adult = yes
        is_cultivator = yes
        OR = {
            has_trait = cultivation_nascent_soul
            has_trait = cultivation_heavenly_being
            has_trait = cultivation_four_axis
            has_trait = cultivation_integration
            has_trait = cultivation_star_shattering
            has_trait = cultivation_sacred_vessel
            has_trait = cultivation_entering_nirvana
            has_trait = cultivation_true_immortal
        }
        NOT = { has_character_flag = cultivation_talent_posture_cooldown }
    }

    immediate = {
        add_character_flag = { flag = cultivation_talent_posture_cooldown days = 365 }

        if = {
            limit = {
                OR = {
                    has_trait = sadistic
                    has_trait = callous
                    has_trait = wrathful
                    has_trait = paranoid
                    has_trait = deceitful
                    has_trait = arbitrary
                    has_trait = ambitious
                    has_trait = vengeful
                }
            }

            every_courtier = {
                limit = {
                    is_alive = yes
                    is_adult = yes
                    has_active_meridians = yes
                    OR = {
                        has_trait = excellent_meridians
                        has_trait = heavenly_meridians
                    }
                    OR = {
                        is_new_cultivator = yes
                        is_known_cultivator = yes
                    }
                }
                add_opinion = {
                    target = scope:root
                    modifier = cultivation_views_me_as_future_threat_opinion
                    years = 10
                }
                # Ruthless high-realm cultivators can eliminate lower-realm threats.
                random_list = {
                    55 = { death = { death_reason = death_murder } }
                    45 = { }
                }
            }

            every_vassal = {
                limit = {
                    is_alive = yes
                    is_adult = yes
                    has_active_meridians = yes
                    OR = {
                        has_trait = excellent_meridians
                        has_trait = heavenly_meridians

                    OR = {
                        is_new_cultivator = yes
                        is_known_cultivator = yes

                    target = scope:root
                    modifier = cultivation_views_me_as_future_threat_opinion
                    years = 10
                }
                random_list = {
                    65 = { death = { death_reason = death_murder } }
                    35 = { }
                }
            }
        }
        else_if = {
            limit = {
                OR = {
                    has_trait = compassionate
                    has_trait = generous
                    has_trait = just
                    has_trait = patient
                    has_trait = humble
                    has_trait = calm
                    has_trait = diligent
                }
            }

            every_courtier = {
                limit = {
                    is_alive = yes
                    is_adult = yes
                    has_active_meridians = yes
                    OR = {
                        has_trait = good_meridians
                        has_trait = excellent_meridians
                        has_trait = heavenly_meridians
                    }
                    NOR = {
                        has_trait = cultivation_nascent_soul
                        has_trait = cultivation_heavenly_being
                        has_trait = cultivation_four_axis
                        has_trait = cultivation_integration
                        has_trait = cultivation_star_shattering
                        has_trait = cultivation_sacred_vessel
                        has_trait = cultivation_entering_nirvana
                        has_trait = cultivation_true_immortal
                    }
                }
                add_opinion = {
                    target = scope:root
                    modifier = cultivation_invests_in_my_talent_opinion
                    years = 10
                }
                add_prestige = 50
            }

            every_vassal = {
                limit = {
                    is_alive = yes
                    is_adult = yes
                    has_active_meridians = yes
                    OR = {
                        has_trait = good_meridians
