cultivation_convert_direct_domain_faith_effect = {
	if = {
		limit = { is_landed = yes }
		# Convert only personally-held county titles (never vassal-held counties).
		every_held_title = {
			limit = { tier = tier_county }
			set_county_faith = root.faith
		}
	}
}
