awaken_meridian_disciples_decision = {
    picture = { reference = "gfx/interface/illustrations/decisions/decision_misc.dds" }
    sort_order = 350
    decision_group_type = major

    title = awaken_meridian_disciples_decision
    desc = awaken_meridian_disciples_decision_desc
    confirm_text = awaken_meridian_disciples_decision_confirm
    selection_tooltip = awaken_meridian_disciples_tooltip

    is_shown = {
        is_cultivator = yes
        OR = { is_ruler = yes has_court_position = court_tutor }
        OR = {
            any_courtier = { is_adult = yes NOT = { is_cultivator = yes } OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians } }
            any_vassal = { is_adult = yes NOT = { is_cultivator = yes } OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians } }
        }
        NOT = { has_character_flag = awaken_meridian_disciples_cooldown }
    }

    is_valid = { always = yes }
    ai_check_interval = 60
    ai_potential = { always = yes }
    ai_will_do = { base = 50 }

    effect = {
        every_courtier = {
            limit = {
                is_adult = yes
                NOT = { is_cultivator = yes }
                OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians }
            }
            add_trait = cultivation_qi_gathering
        }
        every_vassal = {
            limit = {
                is_adult = yes
                NOT = { is_cultivator = yes }
                OR = { has_trait = poor_meridians has_trait = normal_meridians has_trait = good_meridians has_trait = excellent_meridians has_trait = heavenly_meridians }
            }
            add_trait = cultivation_qi_gathering
        }

        add_piety = 150
        add_prestige = 200
        add_character_flag = { flag = awaken_meridian_disciples_cooldown days = 1825 }
    }
}

cleanse_remnant_sects_decision = {
    picture = { reference = "gfx/interface/illustrations/decisions/decision_misc.dds" }
    sort_order = 352
    decision_group_type = major

    title = cleanse_remnant_sects_decision
    desc = cleanse_remnant_sects_decision_desc
    confirm_text = cleanse_remnant_sects_decision_confirm
    selection_tooltip = cleanse_remnant_sects_decision_tooltip

    is_shown = {
        is_cultivator = yes
        is_landed = yes
        any_domain_county = {
            NOT = { faith = root.faith }
        }
        NOT = { has_character_flag = cleanse_remnant_sects_cooldown }
    }

    is_valid = {
        any_domain_county = {
            NOT = { faith = root.faith }
        }
    }

    ai_check_interval = 60
    ai_potential = { always = yes }
    ai_will_do = { base = 20 }

    effect = {
        every_domain_county = {
            limit = {
                NOT = { faith = root.faith }
            }
            faith = { save_temporary_scope_as = former_sect_faith }

            every_vassal = {
                limit = {
                    faith = scope:former_sect_faith
                }
                add_opinion = {
                    target = root
                    modifier = remnant_sect_purged_opinion
                }
            }
        }

        cultivation_convert_direct_domain_faith_effect = yes

        add_piety = 100
        add_character_flag = { flag = cleanse_remnant_sects_cooldown days = 1825 }
    }
}
