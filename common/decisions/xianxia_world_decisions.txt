found_sect_domain_decision = {
    picture = { reference = "gfx/interface/illustrations/decisions/decision_found_holy_order.dds" }
    decision_group_type = major
    sort_order = 521

    title = found_sect_domain_decision
    desc = found_sect_domain_decision_desc
    selection_tooltip = found_sect_domain_decision_tooltip
    confirm_text = found_sect_domain_decision_confirm

    is_shown = {
        is_adult = yes
        is_cultivator = yes
        is_independent_ruler = yes
        OR = {
            highest_held_title_tier = tier_county
            highest_held_title_tier = tier_duchy
        }
    }

    is_valid = {
        realm_size >= 3
        NOT = { has_character_flag = founded_sect_domain }
        OR = {
            highest_held_title_tier = tier_county
            highest_held_title_tier = tier_duchy
        }
    }

    ai_check_interval = 24
    ai_potential = { always = yes }
    ai_will_do = { base = 20 }

    effect = {
        add_character_flag = founded_sect_domain
        if = {
            limit = { NOT = { has_government = clan_government } }
            set_government_type = clan_government
        }
        trigger_event = xianxia_world.1200
    }
}
